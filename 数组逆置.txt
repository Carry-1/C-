void Reverse(int arr[], int n) //数组首地址作为形参
{
	
	int *pointer_1 = arr; //指针pointer_1初始时指向数组起始处，从前往后移动
	int temp; //中间变量
	int *pointer_2 = arr+n-1; //指针pointer_2初始时指向数组结尾处，从后往前移动
	if (n%2 == 0) // 数组长度若为偶数
	{
		while(pointer_1+1 != pointer_2) 
		{//交换
			temp = *pointer_1;
			*pointer_1++ = *pointer_2;
			*pointer_2-- = temp;
		}
	if (n%2 == 1) //数组长度若为奇数
	{
		while(pointer_1 != pointer_2)
		{
			temp = *pointer_1;
			*pointer_1++ = *pointer_2;
			*pointer_2-- = temp;
		}
	}

	}

}
